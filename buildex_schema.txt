#Proposed Schema for CS192 Project: Experiment Builder

TABLE Users:
	uid: int(11) PRIMARY UNSIGNED AUTOINCREMENT
	username: varchar(16) UNIQUE
	password: varchar(128)
	first_name: varchar(32)
	middle_name: varchar(32)
	last_name: varchar(32)
	email_ad: varchar(32)
	temp_password: varchar(128)

TABLE Admins:
	uid: references Users.uid
	aid: int(11) PRIMARY UNSIGNED AUTOINCREMENT

TABLE LaboratoryHeads:
	uid: references Users.uid
	lid: int(11) PRIMARY UNSIGNED AUTOINCREMENT

TABLE Faculty:
	uid: references Users.uid
	fid: int(11) PRIMARY UNSIGNED AUTOINCREMENT
	faculty_num: int(11) UNIQUE
	account_status: boolean #{Admin Confirmed[true], Not Confirmed[false]}

TABLE Graduates:
	uid: references Users.uid
	gid: int(11) PRIMARY UNSIGNED AUTOINCREMENT
	student_num: int(11) UNIQUE
	account_status: boolean #{Admin Confirmed[true], Not Confirmed[false]}

TABLE Laboratories:
	labid: int(11) PRIMARY UNSIGNED AUTOINCREMENT
	name: varchar(32) UNIQUE
	members_count: int(11)

TABLE Experiments:
	eid: int(11) PRIMARY UNSIGNED AUTOINCREMENT
	title: varchar(64)
	category: varchar(32)
	description: varchar(256)
	target_count: int(5) UNSIGNED
	current_count: int(5) UNSIGNED
	status: boolean #{COMPLETE[true],ONGOING[false]}
	is_published: boolean #{Faculty Confirmed[true], Not Confirmed[false]}
	path: varchar(64)

TABLE Respondents:
	rid: int(11) PRIMARY UNSIGNED AUTOINCREMENT
	#OPTIONAL FIELDS:
	first_name: varchar(32)
	middle_name: varchar(32)
	last_name: varchar(32)
	email_ad: varchar(32)
	age: int(2) UNSIGNED
	street_addr: varchar(64)
	barangay_addr: varchar(64)
	city_addr: varchar(64)
	nationality: varchar(32)
	birthdate: Date(10)
	sex: boolean #{MALE[False],FEMALE[True]}
	gender: varchar(32)
	civil_status: int[1] UNSIGNED #{SINGLE[0],MARRIED[1],SEPARATED[2],WIDOWED[3]}

TABLE request:
	fid: int(11) references Faculty.fid
	eid: int(11) references Experiments.eid
	since: Date(10)

TABLE advise:
	fid: int(11) references Faculty.fid
	eid: int(11) references Experiments.eid
	since: Date(10)

TABLE faculty_conduct:
	fid: int(11) references Faculty.fid
	eid: int(11) references Experiments.eid
	since: Date(10)

TABLE graduates_conduct:
	gid: int(11) references Graduates.gid
	eid: int(11) references Experiments.eid
	since: Date(10)


TABLE answer:
	rid: int(11) references Respondents.rid
	eid: int(11) references Experiments.eid
	since: Date(10)

TABLE manages:
	lid: int(11) references LaboratoryHeads.lid
	labid: int(11) references Laboratories.labid
	since: Date(10)

TABLE faculty_member_of:
	fid: references Faculty.fid
	labid: references Laboratories.labid
	since: Date(10)

TABLE graduates_member_of:
	gid: references Graduates.gid
	labid: references Laboratories.labid
	since: Date(10)
